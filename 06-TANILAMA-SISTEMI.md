# 06 - TANILAMA SİSTEMİ
# SPRINTA - Bilimsel Baseline Tanılama Protokolü

## Genel Bakış

Tanılama sistemi, her yeni kullanıcının başlangıç kognitif profilini ölçer.
Bu ölçüm 3 alt teste dayanır ve toplam 10-12 dakika sürer.

**Hedef:** Kullanıcının baseline ARP değerini hesaplayarak zayıf yönlerini
tespit etmek ve kişiselleştirilmiş antrenman planı oluşturmak.

---

## Tanılama Protokolü

### Alt Test 1: Hız Testi (3 dakika)
- 300 kelimelik MEB 8. sınıf düzeyinde metin
- Kullanıcı metni okur, bitince "Okudum" tuşuna basar
- WPM = (300 / okuma_saniye) × 60

### Alt Test 2: Kavrama Testi (4 dakika)
- 500 kelimelik metin (farklı metin)
- 5 çoktan seçmeli soru:
  * Ana fikir (1 soru)
  * Detay (2 soru)
  * Çıkarım (2 soru)
- Comprehension = (doğru_sayı / 5) × 100

### Alt Test 3: Dikkat Testi (3 dakika)
- 60 saniyelik sayaç
- Ekranda dikkat dağıtıcı elementler rastgele belirir
- Kullanıcı dikkat dağıtıcılara DOKUNMAMALI
- Attention Score = 100 - (dokunma_sayısı × 15), minimum 0

## ARP Hesaplama

```typescript
// packages/shared/src/engine/diagnosticCalculator.ts

import { EXAM_ARP_TARGETS } from './arpCalculator';

export interface DiagnosticResult {
  baselineWpm: number;
  comprehensionScore: number;
  attentionScore: number;
  baselineRei: number;
  baselineCsf: number;
  baselineArp: number;
  weaknesses: WeaknessRanking[];
  examProgress: Record<string, number>;
  recommendedModule: string;
}

export interface WeaknessRanking {
  area: 'speed_control' | 'deep_comprehension' | 'attention_power';
  score: number;
  label: string;
  priority: number;
}

export function calculateDiagnosticArp(
  wpm: number,
  comprehension: number,
  attention: number
): DiagnosticResult {
  // REI = sustainable_wpm × (comprehension / 100)
  // Tanılamada sustainable_wpm = measured_wpm (tek ölçüm)
  const rei = wpm * (comprehension / 100);

  // CSF tahmini: dikkat skoru + hız tutarlılığı (tanılamada tek veri noktası)
  // Dikkat skoru CSF'ye dönüştürülür: attention/100 → CSF bileşeni
  const attentionFactor = attention / 100;
  const speedFactor = Math.min(wpm / 300, 1.0); // 300 WPM ideal referans
  const csf = (attentionFactor * 0.6 + speedFactor * 0.4);

  // ARP = REI × CSF
  const arp = Math.round(rei * csf);

  // Zayıflık analizi
  const weaknesses: WeaknessRanking[] = [];

  if (wpm < 180) {
    weaknesses.push({
      area: 'speed_control',
      score: wpm,
      label: 'Okuma Hızı',
      priority: wpm < 120 ? 1 : 2
    });
  }

  if (comprehension < 65) {
    weaknesses.push({
      area: 'deep_comprehension',
      score: comprehension,
      label: 'Kavrama',
      priority: comprehension < 40 ? 1 : 2
    });
  }

  if (attention < 60) {
    weaknesses.push({
      area: 'attention_power',
      score: attention,
      label: 'Dikkat',
      priority: attention < 40 ? 1 : 2
    });
  }

  // Önceliğe göre sırala
  weaknesses.sort((a, b) => a.priority - b.priority);

  // Sınav hedeflerine göre ilerleme
  const examProgress: Record<string, number> = {};
  for (const [exam, target] of Object.entries(EXAM_ARP_TARGETS)) {
    examProgress[exam] = Math.min(Math.round((arp / target.target) * 100), 100);
  }

  // Önerilen modül: en zayıf alan veya varsayılan hız
  const recommendedModule = weaknesses.length > 0
    ? weaknesses[0].area
    : 'speed_control';

  return {
    baselineWpm: wpm,
    comprehensionScore: comprehension,
    attentionScore: attention,
    baselineRei: Math.round(rei),
    baselineCsf: Math.round(csf * 100) / 100,
    baselineArp: arp,
    weaknesses,
    examProgress,
    recommendedModule
  };
}
```

## React Native Tanılama Ekranı

```typescript
// apps/mobile/app/(onboarding)/diagnostic.tsx

import React, { useState, useEffect, useRef } from 'react';
import {
  View, Text, ScrollView, TouchableOpacity,
  StyleSheet, Animated, Alert
} from 'react-native';
import { router } from 'expo-router';
import { useAuthStore } from '../../src/stores/authStore';
import { supabase } from '../../src/lib/supabase';
import { calculateDiagnosticArp, diagnosticQuestions } from '@sprinta/shared';
import { SPEED_TEST_TEXT, COMPREHENSION_TEXT } from '../../src/content/diagnosticTexts';

type Phase =
  | 'intro'
  | 'speed_reading'
  | 'speed_done'
  | 'comprehension_reading'
  | 'comprehension_questions'
  | 'attention_test'
  | 'analyzing'
  | 'results';

export default function DiagnosticScreen() {
  const { student, refreshProfile } = useAuthStore();
  const [phase, setPhase] = useState<Phase>('intro');
  const [currentQuestion, setCurrentQuestion] = useState(0);
  const [answers, setAnswers] = useState<number[]>([]);
  const [readingStartTime, setReadingStartTime] = useState(0);
  const [speedWpm, setSpeedWpm] = useState(0);
  const [attentionScore, setAttentionScore] = useState(100);
  const [attentionTaps, setAttentionTaps] = useState(0);
  const [timeLeft, setTimeLeft] = useState(60);
  const [distractorVisible, setDistractorVisible] = useState(false);
  const [distractorPos, setDistractorPos] = useState({ x: 0, y: 0 });
  const [result, setResult] = useState<any>(null);

  const attentionTimerRef = useRef<ReturnType<typeof setInterval> | null>(null);
  const distractorTimerRef = useRef<ReturnType<typeof setInterval> | null>(null);
  const fadeAnim = useRef(new Animated.Value(0)).current;

  useEffect(() => {
    Animated.timing(fadeAnim, {
      toValue: 1,
      duration: 400,
      useNativeDriver: true
    }).start();
  }, [phase]);

  // ─── PHASE: SPEED READING ───────────────────────────────────────────
  const startSpeedTest = () => {
    setReadingStartTime(Date.now());
    setPhase('speed_reading');
  };

  const finishSpeedReading = () => {
    const elapsed = (Date.now() - readingStartTime) / 1000;
    const wpm = Math.round((300 / elapsed) * 60);
    setSpeedWpm(Math.min(wpm, 600)); // cap at 600
    setPhase('speed_done');
  };

  // ─── PHASE: COMPREHENSION ───────────────────────────────────────────
  const startComprehension = () => {
    setReadingStartTime(Date.now());
    setPhase('comprehension_reading');
  };

  const finishComprehensionReading = () => {
    setPhase('comprehension_questions');
  };

  const answerQuestion = (answerIndex: number) => {
    const newAnswers = [...answers, answerIndex];
    setAnswers(newAnswers);

    if (currentQuestion < diagnosticQuestions.length - 1) {
      setCurrentQuestion(currentQuestion + 1);
    } else {
      // Tüm sorular bitti
      startAttentionTest();
    }
  };

  const getComprehensionScore = () => {
    const correct = answers.filter(
      (ans, i) => ans === diagnosticQuestions[i].correctAnswer
    ).length;
    return Math.round((correct / diagnosticQuestions.length) * 100);
  };

  // ─── PHASE: ATTENTION TEST ──────────────────────────────────────────
  const startAttentionTest = () => {
    setPhase('attention_test');
    setTimeLeft(60);
    setAttentionTaps(0);

    attentionTimerRef.current = setInterval(() => {
      setTimeLeft(prev => {
        if (prev <= 1) {
          clearInterval(attentionTimerRef.current!);
          clearInterval(distractorTimerRef.current!);
          finishAttentionTest();
          return 0;
        }
        return prev - 1;
      });
    }, 1000);

    // Dikkat dağıtıcılar rastgele konumlarda belirir
    distractorTimerRef.current = setInterval(() => {
      const x = 20 + Math.random() * 280;
      const y = 100 + Math.random() * 300;
      setDistractorPos({ x, y });
      setDistractorVisible(true);

      setTimeout(() => setDistractorVisible(false), 800);
    }, 2500 + Math.random() * 1500);
  };

  const handleDistractorTap = () => {
    setAttentionTaps(prev => prev + 1);
    setDistractorVisible(false);
  };

  const finishAttentionTest = () => {
    const score = Math.max(0, 100 - attentionTaps * 15);
    setAttentionScore(score);
    analyzeResults(score);
  };

  // ─── PHASE: ANALYZING ───────────────────────────────────────────────
  const analyzeResults = async (attScore: number) => {
    setPhase('analyzing');
    const comprehension = getComprehensionScore();

    const diagnosticResult = calculateDiagnosticArp(
      speedWpm,
      comprehension,
      attScore
    );

    setResult(diagnosticResult);

    // Veritabanına kaydet
    await saveDiagnostic(diagnosticResult);

    setTimeout(() => setPhase('results'), 2000);
  };

  const saveDiagnostic = async (diagnosticResult: any) => {
    if (!student) return;

    try {
      // Tanılama kaydı
      await supabase.from('diagnostics').insert({
        student_id: student.id,
        speed_wpm: speedWpm,
        comprehension_score: getComprehensionScore(),
        attention_score: diagnosticResult.attentionScore,
        baseline_rei: diagnosticResult.baselineRei,
        baseline_csf: diagnosticResult.baselineCsf,
        baseline_arp: diagnosticResult.baselineArp,
        weaknesses: diagnosticResult.weaknesses,
        recommended_module: diagnosticResult.recommendedModule
      });

      // Öğrenci profilini güncelle
      await supabase
        .from('students')
        .update({
          baseline_arp: diagnosticResult.baselineArp,
          current_arp: diagnosticResult.baselineArp,
          has_completed_diagnostic: true
        })
        .eq('id', student.id);

      // Cognitive profile başlat
      await supabase
        .from('cognitive_profiles')
        .upsert({
          student_id: student.id,
          sustainable_wpm: speedWpm,
          peak_wpm: speedWpm,
          comprehension_baseline: getComprehensionScore(),
          current_rei: diagnosticResult.baselineRei,
          current_csf: diagnosticResult.baselineCsf,
          current_arp: diagnosticResult.baselineArp,
          stability_index: 50 // Başlangıç değeri
        });

      await refreshProfile();
    } catch (error) {
      console.error('Diagnostic save error:', error);
    }
  };

  // ─── RENDER ─────────────────────────────────────────────────────────

  if (phase === 'intro') {
    return (
      <View style={styles.container}>
        <Animated.View style={[styles.content, { opacity: fadeAnim }]}>
          <Text style={styles.emoji}>🧠</Text>
          <Text style={styles.title}>Başlamadan Önce</Text>
          <Text style={styles.subtitle}>
            Sana özel antrenman planı oluşturmak için kısa bir tanılama yapacağız.
          </Text>
          <View style={styles.infoBox}>
            <Text style={styles.infoTitle}>Bu tanılama:</Text>
            <Text style={styles.infoItem}>⏱ Yaklaşık 10-12 dakika sürer</Text>
            <Text style={styles.infoItem}>📖 3 kısa testten oluşur</Text>
            <Text style={styles.infoItem}>📊 Başlangıç seviyeni ölçer</Text>
            <Text style={styles.infoItem}>🎯 Zayıf noktalarını tespit eder</Text>
          </View>
          <Text style={styles.tip}>
            💡 Sakin bir ortamda, telefona odaklanarak yapmanı öneririz.
          </Text>
          <TouchableOpacity style={styles.primaryButton} onPress={startSpeedTest}>
            <Text style={styles.primaryButtonText}>Tanılamayı Başlat →</Text>
          </TouchableOpacity>
        </Animated.View>
      </View>
    );
  }

  if (phase === 'speed_reading') {
    return (
      <View style={styles.container}>
        <View style={styles.header}>
          <Text style={styles.headerStep}>Test 1/3 — Okuma Hızı</Text>
          <Text style={styles.headerHint}>Metni normal hızında oku, bitince "Okudum" tuşuna bas</Text>
        </View>
        <ScrollView style={styles.textContainer}>
          <Text style={styles.passageTitle}>Anadolu'nun Tarihi Zenginlikleri</Text>
          <Text style={styles.passageText}>{SPEED_TEST_TEXT}</Text>
        </ScrollView>
        <TouchableOpacity style={styles.primaryButton} onPress={finishSpeedReading}>
          <Text style={styles.primaryButtonText}>Okudum ✓</Text>
        </TouchableOpacity>
      </View>
    );
  }

  if (phase === 'speed_done') {
    const minutesPerPage = speedWpm > 0 ? Math.round(300 / speedWpm * 10) / 10 : 0;
    return (
      <View style={styles.container}>
        <Animated.View style={[styles.content, { opacity: fadeAnim }]}>
          <Text style={styles.emoji}>⚡</Text>
          <Text style={styles.title}>Hız Testi Tamamlandı!</Text>
          <View style={styles.metricCard}>
            <Text style={styles.metricValue}>{speedWpm}</Text>
            <Text style={styles.metricLabel}>Kelime / Dakika</Text>
          </View>
          <Text style={styles.subtitle}>
            Sırada kavrama testi var. Hazır olduğunda başla.
          </Text>
          <TouchableOpacity style={styles.primaryButton} onPress={startComprehension}>
            <Text style={styles.primaryButtonText}>Kavrama Testine Geç →</Text>
          </TouchableOpacity>
        </Animated.View>
      </View>
    );
  }

  if (phase === 'comprehension_reading') {
    return (
      <View style={styles.container}>
        <View style={styles.header}>
          <Text style={styles.headerStep}>Test 2/3 — Kavrama</Text>
          <Text style={styles.headerHint}>Metni dikkatlice oku, ardından sorular gelecek</Text>
        </View>
        <ScrollView style={styles.textContainer}>
          <Text style={styles.passageTitle}>Türkiye'nin Doğal Çeşitliliği</Text>
          <Text style={styles.passageText}>{COMPREHENSION_TEXT}</Text>
        </ScrollView>
        <TouchableOpacity style={styles.primaryButton} onPress={finishComprehensionReading}>
          <Text style={styles.primaryButtonText}>Soruları Yanıtla →</Text>
        </TouchableOpacity>
      </View>
    );
  }

  if (phase === 'comprehension_questions') {
    const q = diagnosticQuestions[currentQuestion];
    return (
      <View style={styles.container}>
        <View style={styles.header}>
          <Text style={styles.headerStep}>
            Soru {currentQuestion + 1}/{diagnosticQuestions.length}

          <View style={styles.progressBar}>
            <View
              style={[
                styles.progressFill,
                { width: `${((currentQuestion) / diagnosticQuestions.length) * 100}%` }
              ]}
            />
          </View>
        </View>
        <View style={styles.questionContainer}>
          <Text style={styles.questionText}>{q.text}</Text>
          {q.options.map((opt, i) => (

              key={i}
              style={styles.optionButton}
              onPress={() => answerQuestion(i)}

              <Text style={styles.optionLetter}>
                {String.fromCharCode(65 + i)}
              </Text>
              <Text style={styles.optionText}>{opt}</Text>
            </TouchableOpacity>
          ))}
        </View>
      </View>
    );
  }

  if (phase === 'attention_test') {
    return (
      <View style={styles.container}>
        <View style={styles.header}>

        </View>

                <Text style={styles.distractorText}>✨</Text>
              </TouchableOpacity>
            )}
          </View>
          <Text style={styles.attentionHint}>
            Dokunma: {attentionTaps}
          </Text>
        </View>
      </View>
    );
  }

  if (phase === 'analyzing') {
    return (
      <View style={styles.container}>
        <Animated.View style={[styles.content, { opacity: fadeAnim }]}>
          <Text style={styles.emoji}>⚙️</Text>
          <Text style={styles.title}>Analiz Ediliyor...</Text>
          <Text style={styles.subtitle}>
            Performans verilerini işliyoruz ve sana özel profil oluşturuyoruz.
          </Text>
        </Animated.View>
      </View>
    );
  }

  if (phase === 'results' && result) {
    return (
      <ScrollView style={styles.container}>
        <Animated.View style={[styles.content, { opacity: fadeAnim }]}>
          <Text style={styles.emoji}>📊</Text>
          <Text style={styles.title}>Tanılama Tamamlandı!</Text>

          {/* ARP Kartı */}
          <View style={styles.arpCard}>
            <Text style={styles.arpLabel}>Başlangıç ARP Puanın</Text>
            <Text style={styles.arpValue}>{result.baselineArp}</Text>
            <Text style={styles.arpSub}>
              REI: {result.baselineRei} × CSF: {result.baselineCsf}
            </Text>
          </View>

          {/* Metrikler */}
          <View style={styles.metricsRow}>
            <View style={styles.metricItem}>
              <Text style={styles.metricValue2}>{result.baselineWpm}</Text>
              <Text style={styles.metricLabel2}>KPD Hız</Text>
            </View>
            <View style={styles.metricItem}>
              <Text style={styles.metricValue2}>{result.comprehensionScore}%</Text>
              <Text style={styles.metricLabel2}>Kavrama</Text>
            </View>
            <View style={styles.metricItem}>
              <Text style={styles.metricValue2}>{result.attentionScore}%</Text>
              <Text style={styles.metricLabel2}>Dikkat</Text>
            </View>
          </View>

          {/* Sınav Hedefleri */}
          <View style={styles.examSection}>
            <Text style={styles.sectionTitle}>Sınav Hedeflerine Uzaklık</Text>
            {Object.entries(result.examProgress).map(([exam, progress]: [string, any]) => (
              <View key={exam} style={styles.examRow}>
                <Text style={styles.examName}>{exam}</Text>
                <View style={styles.examBar}>
                  <View style={[styles.examFill, { width: `${progress}%` }]} />
                </View>
                <Text style={styles.examPct}>{progress}%</Text>
              </View>
            ))}
          </View>

          {/* Zayıflıklar */}
          {result.weaknesses.length > 0 && (
            <View style={styles.weaknessBox}>
              <Text style={styles.weaknessTitle}>🎯 Öncelikli Gelişim Alanın</Text>
              <Text style={styles.weaknessName}>
                {result.weaknesses[0].area === 'speed_control' && '⚡ Okuma Hızı'}
                {result.weaknesses[0].area === 'deep_comprehension' && '📚 Kavrama'}
                {result.weaknesses[0].area === 'attention_power' && '🎯 Dikkat'}
              </Text>
            </View>
          )}

          <TouchableOpacity
            style={styles.primaryButton}
            onPress={() => router.replace('/(tabs)')}
          >
            <Text style={styles.primaryButtonText}>Antrenmana Başla 🚀</Text>
          </TouchableOpacity>
        </Animated.View>
      </ScrollView>
    );
  }

  return null;
}
```
